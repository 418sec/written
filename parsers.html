Written

With Written, you can cherry-pick which parser you want to use. Every instance below was initialized with different parsers. You can see how easy it is to select what you need.

Markdown Rendered 
// Only use h1, h2, h3, ul as block parsers
// Use all inlines parsers
var parsers = new Written.Parsers({blocks: ['h1', 'h2', 'h3', 'ul']})
new Written(document.querySelector('.editor'), {parsers: parsers})
Markdown Rendered
# Documentation 
// Use all block parsers
// Only use links and strong words as inline parsers
var parsers = new Written.Parsers({inlines: ['a', 'strong']})
new Written(document.querySelector('.editor'), {parsers: parsers})

# Documentation 
// Use no block parser

var parsers = new Written.Parsers({blocks: []})

# Documentation 
// Only use h1, h2, h3 as block parsers
// Only use links, strong and italic as inline parsers
var parsers = new Written.Parsers({blocks: ['h1', 'h2', 'h3'], inlines: ['a', 'strong', 'em']})
new Written(document.querySelector('.editor'), {parsers: parsers})
